## 一、概述

本方案旨在为 Total Chaos 项目添加多语言支持，实现英语（EN）、越南语（VI）、中文（ZH）、俄语（RU）等多语言切换功能，并在右侧导航栏添加语言切换器。

### 技术栈
- **框架**: Next.js 15.3 (App Router)
- **国际化方案**: next-intl (推荐，与 Next.js 15 完美集成)
- **UI框架**: React 18 + TypeScript
- **样式**: Tailwind CSS

---

## 二、技术选型：next-intl

### 为什么选择 next-intl？

1. **Next.js 15 官方推荐**：与 App Router 深度集成
2. **类型安全**：完整的 TypeScript 支持
3. **性能优化**：支持静态生成和服务端渲染
4. **SEO友好**：支持本地化URL和元数据
5. **灵活性**：支持命名空间、嵌套翻译、格式化等

### 替代方案对比

| 方案 | 优点 | 缺点 | 推荐度 |
|------|------|------|--------|
| next-intl | App Router 原生支持，类型安全 | 需要学习新API | ⭐⭐⭐⭐⭐ |
| next-i18next | 社区成熟，文档丰富 | 对 App Router 支持不够完善 | ⭐⭐⭐ |
| react-intl | 功能强大，不依赖框架 | 配置复杂，包体积大 | ⭐⭐⭐ |
| 自研方案 | 完全可控 | 开发成本高，维护困难 | ⭐⭐ |

### 步骤 3：配置 next-intl

#### 3.1 创建 src/i18n/request.ts

```typescript
import { getRequestConfig } from 'next-intl/server';
import { routing } from './routing';

export default getRequestConfig(async ({ requestLocale }) => {
  // 这通常对应于 `[locale]` 段
  let locale = await requestLocale;

  // 确保传入的 `locale` 是有效的
  if (!locale || !routing.locales.includes(locale as any)) {
    locale = routing.defaultLocale;
  }

  return {
    locale,
    messages: (await import(`../messages/${locale}.json`)).default
  };
});
```

#### 3.2 创建 src/i18n/routing.ts

```typescript
import { defineRouting } from 'next-intl/routing';
import { createNavigation } from 'next-intl/navigation';

export const routing = defineRouting({
  // 支持的所有语言
  locales: ['en', 'vi', 'zh', 'ru'],

  // 默认语言
  defaultLocale: 'en',

  // 本地化路径前缀策略
  // 'always': 总是显示语言前缀（推荐）
  // 'as-needed': 默认语言不显示前缀
  localePrefix: 'always'
});
```

#### 3.3 创建 src/i18n/navigation.ts

```typescript
import { createNavigation } from 'next-intl/navigation';
import { routing } from './routing';

export const { Link, redirect, usePathname, useRouter } =
  createNavigation(routing);
```

```typescript
// 根布局不再包含 metadata（移到 [locale]/layout.tsx）
export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {

  const clarityId = process.env.NEXT_PUBLIC_CLARITY_ID;
```

```typescript
  // 优先读取对应语言的文章
  const localizedPath = path.join(
    process.cwd(),
    'src/content',
    locale,  // 新增：根据语言加载不同目录
    ...slug
  ) + '.mdx';

  try {
    const fileContent = await fs.readFile(localizedPath, 'utf8');
    const { data, content } = matter(fileContent);
    return { frontmatter: data, content, locale };
  } catch {
    // 如果找不到对应语言的文章，尝试回退到英语版本

      } catch {
        return null;
      }
    }
    return null;
  }

// 生成所有语言的静态路径
export async function generateStaticParams() {
  const contentDir = path.join(process.cwd(), 'src/content');
  const locales = ['en', 'vi', 'zh', 'ru'];
  const allParams: Array<{ locale: string; slug: string[] }> = [];

  async function getAllMdxFiles(
    dir: string,
    basePath: string[] = []
  ): Promise<string[][]> {
    const entries = await fs.readdir(dir, { withFileTypes: true });
    const paths: string[][] = [];

    for (const entry of entries) {
      const fullPath = path.join(dir, entry.name);

      if (entry.isDirectory()) {
        const subPaths = await getAllMdxFiles(fullPath, [...basePath, entry.name]);
        paths.push(...subPaths);
      } else if (entry.name.endsWith('.mdx')) {
        const fileName = entry.name.replace('.mdx', '');
        paths.push([...basePath, fileName]);
      }
    }

    return paths;
  }

  // 为每种语言生成路径
  for (const locale of locales) {
    const localeContentDir = path.join(contentDir, locale);

    try {
      const allPaths = await getAllMdxFiles(localeContentDir);
      for (const slug of allPaths) {
        allParams.push({ locale, slug });
      }
    } catch (error) {
      // 如果某个语言目录不存在，跳过
      console.warn(`Content directory for ${locale} not found`);
    }
  }

  return allParams;
}
```

#### 10.3 文章内容迁移脚本

为了方便批量迁移现有英文文章，可以创建一个迁移脚本：

**scripts/migrate-content-i18n.js（新建）**

```javascript
const fs = require('fs');
const path = require('path');

const contentDir = path.join(__dirname, '../src/content');
const enDir = path.join(contentDir, 'en');

// 创建英文内容目录
if (!fs.existsSync(enDir)) {
  fs.mkdirSync(enDir, { recursive: true });
}

// 获取所有现有的 MDX 文件（排除已经在语言目录下的）
function getAllMdxFiles(dir, basePath = '') {
  const entries = fs.readdirSync(dir, { withFileTypes: true });
  const files = [];

  for (const entry of entries) {
    // 跳过语言目录
    if (['en', 'vi', 'zh', 'ru'].includes(entry.name)) {
      continue;
    }

    const fullPath = path.join(dir, entry.name);
    const relativePath = path.join(basePath, entry.name);

    if (entry.isDirectory()) {
      files.push(...getAllMdxFiles(fullPath, relativePath));
    } else if (entry.name.endsWith('.mdx')) {
      files.push(relativePath);
    }
  }

  return files;
}

// 迁移文件
function migrateFiles() {
  const files = getAllMdxFiles(contentDir);

  console.log(`Found ${files.length} MDX files to migrate`);

  for (const file of files) {
    const sourcePath = path.join(contentDir, file);
    const targetPath = path.join(enDir, file);
    const targetDir = path.dirname(targetPath);

    // 创建目标目录
    if (!fs.existsSync(targetDir)) {
      fs.mkdirSync(targetDir, { recursive: true });
    }

    // 复制文件到英文目录
    fs.copyFileSync(sourcePath, targetPath);
    console.log(`Migrated: ${file} -> en/${file}`);

    // 更新 canonical URL
    let content = fs.readFileSync(targetPath, 'utf8');
    content = content.replace(
      /canonical: "https:\/\/deadlydelivery\.net\//g,
      'canonical: "https://totalchaos.info/en/'
    );
    fs.writeFileSync(targetPath, content);
  }

  console.log('\n✅ Migration complete!');
  console.log('Next steps:');
  console.log('1. Review migrated files in src/content/en/');
  console.log('2. Create translations in src/content/vi/, zh/, ru/');
  console.log('3. Test the multi-language routing');
}

migrateFiles();
```

### 5.2 URL 重定向策略

为了保持 SEO 权重，需要设置 301 重定向：

**在 middleware.ts 中添加：**

```typescript
import createMiddleware from 'next-intl/middleware';
import { routing } from './i18n/routing';
import { NextRequest, NextResponse } from 'next/server';

const intlMiddleware = createMiddleware(routing);

export default function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // 旧 URL 重定向到新 URL
  // 例如：/codes -> /en/codes
  const oldPaths = [
    '/codes',
    '/classes',
    '/monsters',
    '/guides',
    '/items',
    '/floors',
    '/builds',
    '/wiki',
    '/community',
    '/news',
  ];

  for (const oldPath of oldPaths) {
    if (pathname === oldPath || pathname.startsWith(`${oldPath}/`)) {
      const newUrl = request.nextUrl.clone();
      newUrl.pathname = `/en${pathname}`;
      return NextResponse.redirect(newUrl, 301);
    }
  }

  return intlMiddleware(request);
}

export const config = {
  matcher: [
    '/((?!_next|_vercel|.*\\..*).*)',
    '/',
    '/(en|vi|zh|ru)/:path*'
  ]
};
```

### 6.3 翻译质量保证

创建 `scripts/check-translations.js`：

```javascript
const fs = require('fs');
const path = require('path');

const locales = ['en', 'vi', 'zh', 'ru'];
const messagesDir = path.join(__dirname, '../src/messages');

function getAllKeys(obj, prefix = '') {
  let keys = [];
  for (const key in obj) {
    const newKey = prefix ? `${prefix}.${key}` : key;
    if (typeof obj[key] === 'object' && obj[key] !== null) {
      keys = keys.concat(getAllKeys(obj[key], newKey));
    } else {
      keys.push(newKey);
    }
  }
  return keys;
}

// 读取所有语言文件
const translations = {};
for (const locale of locales) {
  const filePath = path.join(messagesDir, `${locale}.json`);
  translations[locale] = JSON.parse(fs.readFileSync(filePath, 'utf8'));
}

// 获取英文的所有 keys（作为基准）
const baseKeys = new Set(getAllKeys(translations['en']));

// 检查其他语言是否缺失 key
let hasErrors = false;
for (const locale of locales) {
  if (locale === 'en') continue;

  const currentKeys = new Set(getAllKeys(translations[locale]));
  const missingKeys = [...baseKeys].filter(key => !currentKeys.has(key));

  if (missingKeys.length > 0) {
    console.error(`\n❌ Missing keys in ${locale}.json:`);
    missingKeys.forEach(key => console.error(`  - ${key}`));
    hasErrors = true;
  }
}

if (!hasErrors) {
  console.log('\n✅ All translation files are complete!');
} else {
  process.exit(1);
}
```

```typescript
import { MetadataRoute } from 'next';
import { routing } from '@/i18n/routing';

export default function sitemap(): MetadataRoute.Sitemap {
  const baseUrl = 'https://totalchaos.info';
  const pages = [
    '',
    '/codes',
    '/classes',
    '/monsters',
    '/guides',
    '/items',
    '/floors',
    '/builds',
    '/wiki',
    '/community',
  ];

  const sitemap: MetadataRoute.Sitemap = [];

  for (const locale of routing.locales) {
    for (const page of pages) {
      sitemap.push({
        url: `${baseUrl}/${locale}${page}`,
        lastModified: new Date(),
        changeFrequency: 'daily',
        priority: page === '' ? 1.0 : 0.8,
        alternates: {
          languages: Object.fromEntries(
            routing.locales.map((loc) => [
              loc,
              `${baseUrl}/${loc}${page}`
            ])
          )
        }
      });
    }
  }

  return sitemap;
}
```
